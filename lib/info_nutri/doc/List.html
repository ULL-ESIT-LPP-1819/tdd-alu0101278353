def clasificar_imc

        n=@head
        while n!=nil do
                if n.value.imc >=30
                        obesidad << n.value
                else
                        sin_obesidad << n.value
                end
                
                n=n.next
        end

end

def extraer_head ()
        if @head==@tail
                @head=nil
                @tail=nil
        else
                @head=@head.next
                @head.prev=nil

        end
end

def insertar_head (value)
        if self.empty
                n=Node.new(value,nil,nil)
                @head=n
                @tail=n
        else
                n=Node.new(value,@head,nil)
                @head.prev=n
                @head=n
        end
end

def insertar_tail (value)
        if self.empty
                n=Node.new(value,nil,nil)
                @tail=n
                @head=n
        else
                n=Node.new(value,nil,@tail)
                @tail.next=n
                @tail=n
        end
end
